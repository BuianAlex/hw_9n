POST /login 
req
 {"loginName": String "password": String}
res 200
  {
    "result": [
      {  name: String,
        loginName: String,
        email: String,
        phone: String,
        photo: String,
        usergroup: String,
      }
    ] 
  }
res error
  {
    "status": 401,
    "message": `Sorry, the member name and password
    you entered do not match. Please try again` 
  }
or
  {
    "status": 400,
    "message": "Invalid data"
  }
